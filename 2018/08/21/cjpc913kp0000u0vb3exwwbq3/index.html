实现目标div在另一个div中的拖拽效果

最近项目中遇到的一个小功能，总结了一下，感觉还不错.
github地址： https://github.com/Abriams/js-drafting

<!DOCTYPE html>
    <html>
    <head>
        <meta charset="utf-8" />
        <meta http-equiv="X-UA-Compatible" content="IE=edge">
        <title>Page Title</title>
        <meta name="viewport" content="width=device-width, initial-scale=1">
        <style>
            *{
                margin: 0;
                padding: 0;
            }
            #draft_big {  
                border: 1px solid #FF3300;  
                width: 300px;  
                height: 300px;  
                position: relative;  
                margin-left: 400px;
                margin-top: 400px;
            }  
            #draft_small {  
                background: #99CC00;  
                width: 50px;  
                height: 50px;  
                position: absolute;  
                cursor: pointer;  
                right: -80px;
            }
        </style>
    </head>
    <body>  
        <div id="draft_big">  
            <div id="draft_small"></div>  
        </div>  
    </body>  
    <script language="javascript">  
        window.onload = function () {
            var draftBig = document.getElementById("draft_big");  
            var draftSmall = document.getElementById("draft_small");  
            draftSmall.onmousedown = function (event) {
                var e = e || window.event;  
                //记录并保存可拖拽div（id为draftSmall的盒子）拖拽前的坐标 
                var startX = e.clientX - draftSmall.offsetLeft;
                var startY = e.clientY - draftSmall.offsetTop;
                // offsetHeight,包括元素的边框、内边距和元素的水平滚动条（如果存在且渲染的话）;
                // offsetLeft,目标元素距离最近定位的父级元素左边的距离；
                // offsetTop,目标元素距离最近定位的父级元素上边的距离；
                // clientX 事件属性返回当事件被触发时鼠标指针向对于浏览器页面（或客户区）的水平坐标。
                // clientY 事件属性返回当事件被触发时鼠标指针向对于浏览器页面（或客户区）的垂直坐标。
                /*鼠标的移动事件*/  
                document.onmousemove = function(e) { 
                    //判断被拖拽的盒子（id为draftSmall的盒子）必须拖拽到目标盒子（id为draftBig盒子中）才能移动；
                    if(e.clientX - 400 < draftBig.offsetHeight && e.clientX > 400) {
                        if(e.clientY -400 < draftBig.offsetWidth && e.clientY > 400) {
                            var e = e || window.event;  
                            //计算当前拖拽的盒子（id为draftSmall的盒子）在目标盒子（id为draftBig盒子）中的坐标；
                            draftSmall.style.left = e.clientX - startX + "px";  
                            draftSmall.style.top = e.clientY - startY + "px";  
                            /*判断拖拽边界（id为draftSmall的盒子四个边界的判断）*/  
                            if (e.clientX - startX <= 0) {  
                                draftSmall.style.left = 0 + "px";  
                            }  
                            if (e.clientY - startX <= 0) {  
                                draftSmall.style.top = 0 + "px";  
                            }  
                            if (e.clientX - startX >= 250) {  
                                draftSmall.style.left = 250 + "px";  
                            }  
                            if (e.clientY - startX >= 250) {  
                                draftSmall.style.top = 250 + "px";  
                            }  
                        }
                    }
                };  
                /*鼠标的抬起事件,终止拖动*/  
                document.onmouseup = function() {  
                    document.onmousemove = null;  
                    document.onmouseup = null;  
                };  
            };
        }; 
    </script>
    </html>
